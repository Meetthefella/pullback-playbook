<script>
function extract(){
let raw=document.getElementById("watchlistRaw").value
let tickers=(raw.match(/\\b[A-Z]{1,5}\\b/g)||[])
document.getElementById("watchlist").value=[...new Set(tickers)].join(", ")
}

function generate(){
let watch=document.getElementById("watchlist").value.split(",").join("\\n")
let acc=document.getElementById("accountSize").value
let risk=document.getElementById("maxRisk").value

let prompt=`Analyse these stocks for my Quality Pullback strategy.

Rules:
- Prefer strong stocks in an uptrend
- Pullback near 20MA or 50MA
- Must stabilise or bounce
- Previous swing high = first target

Account:
£${acc} account
£${risk} max risk

Watchlist:
${watch}

Return:
Watch / Near Pullback / Entry / Avoid
with a pullback score out of 10.`

document.getElementById("prompt").value=prompt
}

function calc(){
let entry=parseFloat(document.getElementById("entry").value)
let stop=parseFloat(document.getElementById("stop").value)
let target=parseFloat(document.getElementById("target").value)
let risk=parseFloat(document.getElementById("maxRisk").value)

if(!entry||!stop){return}

let riskPerShare=entry-stop
let shares=Math.floor(risk/riskPerShare)

document.getElementById("result").innerText=
"Risk per share: "+riskPerShare.toFixed(2)+
" | Shares: "+shares
}
